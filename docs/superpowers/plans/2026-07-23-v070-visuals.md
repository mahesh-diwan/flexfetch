```rust
pub fn gradient_text(text: &str, colors: &[[u8; 3]]) -> String {
    if colors.is_empty() || text.is_empty() {
        return text.to_string();
    }
    let mut result = String::with_capacity(text.len() * 20);
    for (i, ch) in text.char_indices() {
        let color = colors[i % colors.len()];
        result.push_str(&format!(
            "\x1b[38;2;{};{};{}m{}\x1b[0m",
            color[0], color[1], color[2], ch
        ));
    }
    result
}
```

- [ ] **Step 1: Add progress_bar helper in template.rs**

```rust
pub fn progress_bar(percent: u8, width: usize, colors: &ThemeColors) -> String {
    let filled = (percent as usize * width) / 100;
    let empty = width - filled;
    let color = if percent < 60 {
        "\x1b[32m" // green
    } else if percent < 85 {
        "\x1b[33m" // yellow
    } else {
        "\x1b[31m" // red
    };
    format!(
        "{}[{}{}]\x1b[0m",
        color,
        "█".repeat(filled),
        "░".repeat(empty),
    )
}
```

struct BoxChars {
    header: &'static str,
    row: &'static str,
    separator: &'static str,
}

fn get_box_chars(style: &str) -> BoxChars {
    match style {
        "double" => BoxChars { header: "╔═", row: "║", separator: "══" },
        "dotted" => BoxChars { header: "┆─", row: "┆", separator: "┄┄" },
        "thick"  => BoxChars { header: "┣━", row: "┃", separator: "━━" },
        "ascii"  => BoxChars { header: "+-", row: "|", separator: "--" },
        _        => BoxChars { header: "╭─", row: "│", separator: "──" },
    }
}
```

pub fn frame_wrap(text: &str, style: &str, color: &str) -> String {
    let (tl, tr, bl, br, h, v) = match style {
        "double" => ("╔", "╗", "╚", "╝", "═", "║"),
        "decorative" => ("┌", "┐", "└", "┘", "─", "│"),
        _ => ("┌", "┐", "└", "┘", "─", "│"),
    };
    let max_w = text.lines().map(|l| l.len()).max().unwrap_or(40);
    let mut result = String::new();
    result.push_str(&format!("{}{}{}\n", color, tl, h.repeat(max_w + 2)));
    for line in text.lines() {
        result.push_str(&format!("{}{} {: <width$} {}\n", color, v, line, v, width = max_w));
    }
    result.push_str(&format!("{}{}{}{}\n", color, bl, h.repeat(max_w + 2), "\x1b[0m"));
    result
}
```

- [ ] **Step 2: Apply frame if configured**

}
